AsianFloatingStrikeCall & AsianFl-BskCall & $\max\left(0, \overline{S}(t) - \frac{1}{t}\sum_{k=1}^t \overline{S}(\tau_k)\right)$ where $\overline{S}(\tau_k) = \frac{1}{d}\sum_{i=1}^d S_i(\tau_k)$ \\

AsianFloatingStrikePut & AsianFl-BskPut & $\max\left(0, \frac{1}{t}\sum_{k=1}^t \overline{S}(\tau_k) - \overline{S}(t)\right)$ where $\overline{S}(\tau_k) = \frac{1}{d}\sum_{i=1}^d S_i(\tau_k)$ \\

\multicolumn{3}{c}{\textbf{MAXDISPERSION \& DISPERSION BASKET OPTIONS (4)}} \\
\midrule

MaxDispersionPut & MaxDisp-BskPut & $\max\left(0, K - \left[\max_i S_i(t) - \min_i S_i(t)\right]\right)$ \\
DispersionCall & Disp-BskCall & $\max\left(0, \sigma(t) - K\right)$ where $\sigma(t) = \sqrt{\frac{1}{d}\sum_{i=1}^d \left(S_i(t) - \overline{S}(t)\right)^2}$ and $\overline{S}(t) = \frac{1}{d}\sum_{i=1}^d S_i(t)$ \\
DispersionPut & Disp-BskPut & $\max\left(0, K - \sigma(t)\right)$ where $\sigma(t) = \sqrt{\frac{1}{d}\sum_{i=1}^d \left(S_i(t) - \overline{S}(t)\right)^2}$ and $\overline{S}(t) = \frac{1}{d}\sum_{i=1}^d S_i(t)$ \\
\midrule
\multicolumn{3}{c}{\textbf{RANK-BASED BASKET OPTIONS (4)}} \\
\midrule
BestOfKCall & BestK-BskCall & $\max\left(0, \frac{1}{k}\sum_{i=1}^k S_{(i)}(t) - K\right)$ where $S_{(1)} \geq S_{(2)} \geq \ldots \geq S_{(d)}$ are sorted prices $S_i(t)$ \\
WorstOfKPut & WorstK-BskPut & $\max\left(0, K - \frac{1}{k}\sum_{i=d-k+1}^d S_{(i)}(t)\right)$ where $S_{(1)} \geq \ldots \geq S_{(d)}$ are sorted prices $S_i(t)$ \\

RankWeightedBasketCall & Rank-BskCall & $\max\left(0, \sum_{i=1}^k w_i S_{(i)}(t) - K\right)$ where $w_i$ are user-specified weights (summing to 1) applied to the $k$ best performing assets, $S_{(1)} \geq S_{(2)} \geq \ldots \geq S_{(d)}$ are sorted prices $S_i(t)$ \\
RankWeightedBasketPut & Rank-BskPut & $\max\left(0, K - \sum_{i=1}^k w_i S_{(i)}(t)\right)$ where $w_i$ are user-specified weights (summing to 1) applied to the $k$ best performing assets, $S_{(1)} \geq \ldots \geq S_{(d)}$ are sorted prices $S_i(t)$ \\

\bottomrule
